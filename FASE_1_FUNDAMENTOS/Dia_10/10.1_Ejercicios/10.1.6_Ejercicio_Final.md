## üß† FRASE CLAVE

## ‚úÖ C√ìDIGO COMPLETO
```python
# 1Ô∏è‚É£ Creo lista vac√≠a para usuarios
usuarios = []  # Caja de fichas

# 2Ô∏è‚É£ Pido cu√°ntos usuarios quiere guardar
n = int(input("¬øCu√°ntos usuarios quieres a√±adir? "))

# 3Ô∏è‚É£ Bucle para pedir datos
for i in range(n):
    nombre = input("Nombre: ")
    edad = int(input("Edad: "))
    ciudad = input("Ciudad: ")

    perfil = {  # Ficha de usuario
        "nombre": nombre,
        "edad": edad,
        "ciudad": ciudad
    }

    usuarios.append(perfil)  # Guardo ficha en caja

# 4Ô∏è‚É£ Mostrar perfiles completos
print("\nPerfiles completos:")
for usuario in usuarios:
    print(usuario)
# Analog√≠a: saco cada ficha y la leo entera

# 5Ô∏è‚É£ Mostrar solo nombres
print("\nSolo nombres:")
for usuario in usuarios:
    print(usuario["nombre"])
# Analog√≠a: saco ficha y miro solo el nombre

# 6Ô∏è‚É£ Contar mayores de edad
mayores = 0
for usuario in usuarios:
    if usuario["edad"] >= 18:
        mayores += 1

print("\nMayores de edad:", mayores)
# Analog√≠a: reviso cada ficha y sumo si cumple

# 7Ô∏è‚É£ Lista de frutas
frutas = [
    {"nombre": "Manzana", "precio": 1.2},
    {"nombre": "Pl√°tano", "precio": 0.8},
    {"nombre": "Naranja", "precio": 1.0}
]

print("\nLista de frutas:")
for fruta in frutas:
    print(fruta["nombre"], "-", fruta["precio"], "‚Ç¨/unidad")
# Analog√≠a: otra caja, pero cada ficha es una fruta
```
